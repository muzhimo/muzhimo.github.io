基于LCD触摸屏的相册播放

发表于 2017-09-23 | 分类于 技术 |

功能说明

  • 在相册开始界面有两个按钮，一个是自动播放，另一个是手动播放。
  • 点击自动播放，按顺序逐渐显示图片，在最后一张显示点击屏幕继续播放，右下角有一个退出键，点击返回开始界面。
  • 点击手动播放显示第一张图片，然后向左滑动显示下一张，向右滑动显示上一张，向上滑动返回开始界面。
  • 图片显示有两种特效，分别是百叶窗和左插入。
  • 小于屏幕的图片居中显示，大于屏幕的图片截图显示。

代码

#include <stdio.h>
#include "jpeglib.h"
#include <setjmp.h>
#include <sys/types.h>
#include <sys/stat.h>
#include <fcntl.h>
#include <unistd.h>
#include <sys/mman.h>
#include <stdlib.h>
#include <linux/input.h>
#include <tslib.h>

unsigned int  *lcd;
int x1_x,x1_y,x2_x,x2_y;
int distance_x = 0;
int distance_y = 0;
int flag = 0;

struct my_error_mgr 
{
    struct jpeg_error_mgr pub;    /* "public" fields */

    jmp_buf setjmp_buffer;    /* for return to caller */
};

typedef struct my_error_mgr * my_error_ptr;

METHODDEF(void)
my_error_exit (j_common_ptr cinfo)
{
    /* cinfo->err really points to a my_error_mgr struct, so coerce pointer */
    my_error_ptr myerr = (my_error_ptr) cinfo->err;

    /* Always display the message. */
    /* We could postpone this until after returning, if we chose. */
    (*cinfo->err->output_message) (cinfo);

    /* Return control to the setjmp point */
    longjmp(myerr->setjmp_buffer, 1);
}

GLOBAL(int)
read_JPEG_file (char * filename,int swit)
{
    struct jpeg_decompress_struct cinfo;
    struct my_error_mgr jerr;
    FILE * infile;        
    JSAMPARRAY buffer;        
    int row_stride;        

    if ((infile = fopen(filename, "rb")) == NULL) 
    {
        fprintf(stderr, "can't open %s\n", filename);
        return 0;
    }

    cinfo.err = jpeg_std_error(&jerr.pub);
    jerr.pub.error_exit = my_error_exit;

    if (setjmp(jerr.setjmp_buffer)) 
    {
        jpeg_destroy_decompress(&cinfo);
        fclose(infile);
        return 0;
    }

    jpeg_create_decompress(&cinfo);

    jpeg_stdio_src(&cinfo, infile);

    (void) jpeg_read_header(&cinfo, TRUE);

    (void) jpeg_start_decompress(&cinfo); 

    row_stride = cinfo.output_width * cinfo.output_components;

    buffer = (*cinfo.mem->alloc_sarray)
        ((j_common_ptr) &cinfo, JPOOL_IMAGE, row_stride, 1);

    char red;  
    char green;
    char blue;
    char a;

    int color;
    int x=0;
    int y=0;
    int height_y=0;
    int width_x=0;
    int newset=0;
    int buf32[cinfo.output_height][cinfo.output_width];

    if((cinfo.output_height<480) && (cinfo.output_width<800))
    {
        x = (800-cinfo.output_width)/2;
        y = (480-cinfo.output_height)/2;
        newset = x+y*800;
        height_y = cinfo.output_height;
        width_x = cinfo.output_width;
    }

    if((cinfo.output_height>=480) && (cinfo.output_width<800))
    {
        x = (800-cinfo.output_width)/2;
        y = 0;
        newset = x+y*800;
        height_y = 480;
        width_x = cinfo.output_width;
    }

    if((cinfo.output_height<480) && (cinfo.output_width>=800))
    {
        x = 0;
        y = (480-cinfo.output_height)/2;
        newset = x+y*800;
        height_y = cinfo.output_height;
        width_x = 800;
    }

    if((cinfo.output_height>=480) && (cinfo.output_width>=800))
    {
        x =0;
        y =0;
        height_y = 480;
        width_x = 800;
    }

    while (cinfo.output_scanline < cinfo.output_height)
    {
        y = cinfo.output_scanline;
        //把一行JPEG数据转换成BMP数据并保存到 buffer中
        (void) jpeg_read_scanlines(&cinfo, buffer, 1);

        char *p =  buffer[0];

        for(x=0;x<cinfo.output_width;x++)
        {
            //判断JPEG图片的位深  
            if(cinfo.output_components == 3)
            {
                a = 0;
            }
            else
            {
                a = *p++;
            }

            red = *p++;
            green =*p++;
            blue =*p++;

            color =  a<< 24 | red <<16 | green << 8 | blue;

            buf32[y][x] = color;
        }
    }

    int i,j;
    if(swit == 0)
    {
        for(i=0;i<height_y;i++)
        {
            for(j=0;j<width_x;j++)
            {
            *(lcd + newset + j +i*800) = buf32[i][j];  
            }
        }
    }

    else if(swit == 1)
    {
        int t = 0;
        int hei_sign = height_y/3;
        int tick = 0;

        for(t = 0 ; t < hei_sign ; t = t + 1)
        {
            if(i + 2 > hei_sign)
                i  = hei_sign;
            for(i=0;i < t;i ++)
            {  
                for(j=0;j<width_x;j++)
                {            
                    *(lcd + newset + j + i*800) = buf32[i][j];  
                    *(lcd + newset + j + (hei_sign-1+i)*800) = buf32[hei_sign-1+i][j]; 
                    *(lcd + newset + j + (2*hei_sign-1+i)*800) = buf32[2*hei_sign-1+i][j];                           
                }  
            }

        }
        for(i=0;i<height_y;i++)
        {
            for(j=0;j<width_x;j++)
            {
                *(lcd + newset + j +i*800) = buf32[i][j];  
            } 
        }
    }

    else if(swit == 2)
    {
        int k=0;
        for(k=0;k<width_x;k++)
        {
            for(i=0;i<k;i++)
            {
                for(j=0;j<height_y;j++)
                {
                    *(lcd + newset + i +j*800) = buf32[j][width_x-k+i];  
                }
            }
        }
    }

    /* Step 7: Finish decompression */

    (void) jpeg_finish_decompress(&cinfo);

    jpeg_destroy_decompress(&cinfo);

    fclose(infile);

    return 1;
}

char lcd_touch()
{
    //打开LCD设置
    struct tsdev* tsfd = ts_open("/dev/event0",0);

    struct ts_sample ts_data;

    //配置LCD设备
    ts_config(tsfd);

    char type;

    while(1)
    {
        ts_read(tsfd,&ts_data,1); //要是0不阻塞 不按都会显示

        //获取X1，Y1
        if(ts_data.pressure > 200 && flag == 0)
        {
            x1_x = ts_data.x;
            x1_y = ts_data.y;
            flag = 1;
        }
        //获取X2，Y2
        if(ts_data.pressure < 10 && flag == 1)
        {
            x2_x = ts_data.x;
            x2_y = ts_data.y;
            flag = 0;

            distance_x = x2_x - x1_x;
            distance_y = x2_y - x1_y;

            if(distance_x > 100)
            {
                //printf("--------->\n");
                type = 3;   //右滑
                break;
            }

            else if(distance_x < -100)
            {
                //printf("<---------\n");
                type = 2;    //左滑
                break;
            }

            else if(distance_x == 0 && distance_y == 0)
            {
                //printf("Click \n"); //点击
                type = 1;
                break;
            }

            if(distance_y < -100)
            {
                //printf("Up \n");
                type = 4;
                break;
            }
        }
    }

    ts_close(tsfd);
    tsfd = NULL;

    return type;    //返回type给用户使用
}

void lcdClick()
{
    //打开LCD设置
    struct tsdev* tsfd = ts_open("/dev/event0",0);

    struct ts_sample ts_data;

    //配置LCD设备
    ts_config(tsfd);

    while(1)
    {
        ts_read(tsfd,&ts_data,1); //要是0不阻塞 不按都会显示

        if(ts_data.pressure > 200)
        {
            x1_x = ts_data.x;
            x1_y = ts_data.y;
            break;
        }
    }

    ts_close(tsfd);
    tsfd = NULL;
}

void playA()
{    
    int x=0;

    for(x=0;x<800*480;x++)
    {
        *(lcd+x) = 0xffffff;
    }

    while(1)
    {
        read_JPEG_file("./图片1.jpg",1);
        sleep(2);

        read_JPEG_file("./图片2.jpg",2);
        sleep(2);

        read_JPEG_file("./图片3.jpg",1);
        sleep(2);

        read_JPEG_file("./结束.jpg",0);

        lcdClick();
        if((x1_x>700) && (x1_x<780) && (x1_y>400) && (x1_y<460))
        {
            return;
        }
        else
        {
            continue;
        }
    }
}

void playM()
{
    int i=0;
    char type = 0;

    read_JPEG_file("./图片1.jpg",1);
    i++;

    while(1)
    {
        type = lcd_touch();

        if(type == 2)    //左滑
        {
            //让I复位
            if(i>=3)
            {
                i=1;
            }
            else
            {
                i++;
            }
        }

        if(type == 3)    //右滑
        {
            if(i<=0)
            {
                i=3;
            }
            else
            {
                i--;
            }
        }

        if(type == 4)    //上滑退出
        {
            return;
        }

        switch(i)
        {
            case 1:
                read_JPEG_file("./图片1.jpg",1);
                break;
            case 2:
                read_JPEG_file("./图片2.jpg",2);
                break;
            case 3:
                read_JPEG_file("./图片3.jpg",1);
                break;
            default:
                break;
        }
    }
}

int main ()
{
    //1.打开LCD设备  
    int lcdfd = open("/dev/fb0",O_RDWR);
    if(lcdfd < 0 )
    printf("open  lcd fail\n");        
    //映射设备  
    lcd=mmap(NULL, 800*480*4,PROT_READ|PROT_WRITE,MAP_SHARED,lcdfd,0);

    char type=0;

    while(1)
    {
        read_JPEG_file("./开始.jpg",0);

        lcdClick();

        if((x1_x>100) && (x1_x<300) && (x1_y>200) && (x1_y<280))
        {
            playA();
        }

        if((x1_x>500) && (x1_x<700) && (x1_y>200) && (x1_y<280))
        {
            playM();
        }
    }

    close(lcdfd);

    return 0;
}
